    public static boolean unique1(int[] data, int timesToRun) {
        int n = data.length;
        for (int nthrun = 0; nthrun < timesToRun; nthrun++)
            for (int j=0; j < n-1; j++)
                for (int k=j+1; k < n; k++)
                    if (data[j] == data[k])
                        return false;            // found duplicate pair
        return true;                         // if we reach this, elements are unique
    }


    public static int[] generateSequentialIntArray(int size) {
        int[] array = new int[size];
        for (int i = 0; i < size; i++) {
            array[i] = i;
        }
        return array;
    }
